       ```python
       # Traffic KPI: active roadworks count + flow status
       try:
           # Count active roadworks (Autobahn features with type="roadwork" or type="closure")
           roadworks_result = await db.execute(text("""
               SELECT COUNT(*) FROM features
               WHERE domain = 'traffic'
               AND properties->>'type' IN ('roadwork', 'closure')
               AND ST_Intersects(geometry, ST_MakeEnvelope(:lon_min, :lat_min, :lon_max, :lat_max, 4326))
           """), {"lon_min": bbox.lon_min, "lat_min": bbox.lat_min, "lon_max": bbox.lon_max, "lat_max": bbox.lat_max})
           active_roadworks = roadworks_result.scalar() or 0

           # Flow status: average congestion from latest readings
           flow_result = await db.execute(text("""
               SELECT congestion_level, COUNT(*) as cnt
               FROM traffic_readings r
               INNER JOIN features f ON f.id = r.feature_id
               WHERE f.domain = 'traffic' AND r.time > NOW() - INTERVAL '1 hour'
               AND ST_Intersects(f.geometry, ST_MakeEnvelope(:lon_min, :lat_min, :lon_max, :lat_max, 4326))
               GROUP BY congestion_level ORDER BY cnt DESC LIMIT 1
           """), {"lon_min": bbox.lon_min, "lat_min": bbox.lat_min, "lon_max": bbox.lon_max, "lat_max": bbox.lat_max})
           flow_row = flow_result.first()
           flow_status_map = {"free": "normal", "moderate": "elevated", "congested": "congested"}
           flow_status = flow_status_map.get(flow_row[0]) if flow_row else None

           traffic_kpi = TrafficKPI(active_roadworks=active_roadworks, flow_status=flow_status, last_updated=...)
       except Exception:
           traffic_kpi = TrafficKPI(active_roadworks=0, flow_status=None, last_updated=None)
       ```

    2. In `backend/app/routers/kpi.py`, add EnergyKPI query:
       ```python
       # Energy KPI: renewable %, generation mix, wholesale price
       try:
           # Latest generation mix from energy_readings (one row per source_type)
           mix_result = await db.execute(text("""
               SELECT DISTINCT ON (source_type) source_type, value_kw, time
               FROM energy_readings
               WHERE time > NOW() - INTERVAL '1 hour'
               ORDER BY source_type, time DESC
           """))
           mix_rows = mix_result.fetchall()
           generation_mix = {row.source_type: row.value_kw / 1000 for row in mix_rows if row.source_type != "price"}
           # Wholesale price
           price_row = next((r for r in mix_rows if r.source_type == "price"), None)
           wholesale_price = price_row.value_kw if price_row else None  # stored as EUR/MWh in value_kw
           # Renewable %
           renewable_sources = {"solar", "wind_onshore", "wind_offshore", "hydro", "biomass"}
           total = sum(generation_mix.values()) if generation_mix else 0
           renewable = sum(v for k, v in generation_mix.items() if k in renewable_sources)
           renewable_pct = (renewable / total * 100) if total > 0 else None

           energy_kpi = EnergyKPI(renewable_percent=renewable_pct, generation_mix=generation_mix, wholesale_price_eur_mwh=wholesale_price, last_updated=...)
       except Exception:
           energy_kpi = EnergyKPI(renewable_percent=None, generation_mix={}, wholesale_price_eur_mwh=None, last_updated=None)
       ```

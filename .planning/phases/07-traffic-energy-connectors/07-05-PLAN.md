```typescript
'use client'
import { Source, Layer } from 'react-map-gl/maplibre'
import type { CircleLayerSpecification } from 'react-map-gl/maplibre'
import { useLayerData } from '@/hooks/useLayerData'
import { FeaturePopup } from './FeaturePopup'

const circleLayer: CircleLayerSpecification = {
  id: 'aqi-circles',
  type: 'circle',
  source: 'aqi',
  paint: {
    'circle-color': ['get', 'aqi_color'],
    'circle-radius': 8,
    'circle-stroke-width': 1,
    'circle-stroke-color': '#ffffff',
  },
}

export function AQILayer({ town, visible, timestamp }: LayerProps) {
  const { data } = useLayerData('air_quality', town, timestamp)
  if (!visible || !data) return null
  return (
    <Source id="aqi" type="geojson" data={data}>
      <Layer {...circleLayer} />
    </Source>
  )
}
```

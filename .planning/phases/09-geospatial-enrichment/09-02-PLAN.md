Create `frontend/components/map/BuildingsLayer.tsx`:

The PMTiles base map (Protomaps) includes OSM building polygons in the `buildings` source layer. MapLibre fill-extrusion can render these with height. This avoids needing a separate WFS download from LGL.

```typescript
'use client';
import { Layer } from 'react-map-gl/maplibre';
import type { FillExtrusionLayerSpecification } from 'react-map-gl/maplibre';

interface BuildingsLayerProps {
  visible: boolean;
}

export default function BuildingsLayer({ visible }: BuildingsLayerProps) {
  // Protomaps PMTiles includes 'buildings' source-layer with 'height' property from OSM
  // fill-extrusion renders 3D building volumes
  const layerSpec: FillExtrusionLayerSpecification = {
    id: 'buildings-3d',
    type: 'fill-extrusion',
    source: 'protomaps',
    'source-layer': 'buildings',
    paint: {
      'fill-extrusion-color': '#d4c4a8',
      'fill-extrusion-height': [
        'coalesce',
        ['get', 'height'],
        10, // Default 10m for buildings without height data
      ],
      'fill-extrusion-base': 0,
      'fill-extrusion-opacity': visible ? 0.7 : 0,
    },
    minzoom: 14,
  };

  return <Layer {...layerSpec} />;
}
```

IMPORTANT: Verify the exact source-layer name in the PMTiles. Protomaps basemaps use 'buildings' as the source-layer name. The 'height' property comes from OSM `building:height` or `height` tags. Many buildings lack height data, so the default fallback of 10m is essential.

If `FillExtrusionLayerSpecification` is not exported from react-map-gl/maplibre, use `maplibre-gl` import instead: `import type { FillExtrusionLayerSpecification } from 'maplibre-gl'`.

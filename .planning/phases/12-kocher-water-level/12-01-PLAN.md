@dataclass
class Observation:
    feature_id: str
    domain: str       # "water" for this connector
    values: dict      # {"level_cm": float, "flow_m3s": float}
    timestamp: datetime | None = None
    source_id: str | None = None

class BaseConnector(ABC):
    async def fetch(self) -> Any: ...
    def normalize(self, raw: Any) -> list[Observation]: ...
    async def run(self) -> None: ...
    async def persist(self, observations: list[Observation]) -> None: ...
    async def upsert_feature(self, source_id, domain, geometry_wkt, properties) -> str: ...
    async def _update_staleness(self) -> None: ...
